# E-commerce DB Design

## 1. DDL

```
CREATE TABLE CATEGORY (
CATEGORY_ID INTEGER AUTO_INCREMENT,
CATEGORY_NAME VARCHAR(255) NOT NULL,
CONSTRAINT CATEGORY_PK PRIMARY KEY (CATEGORY_ID)
);

CREATE TABLE PRODUCTS (
PRODUCT_ID INTEGER AUTO_INCREMENT,
CATEGORY_ID INTEGER NOT NULL,
NAME VARCHAR(255) NOT NULL,
DESCRIPTION VARCHAR(4000) NOT NULL,
PRICE DECIMAL(9,2) NOT NULL CHECK(PRICE > 0),
STOCK_QUANTITY INTEGER NOT  NULL CHECK(STOCK_QUANTITY >= 0),
CONSTRAINT PRODUCT_PK PRIMARY KEY(PRODUCT_ID),
CONSTRAINT PRODUCT_CATEGORY_FK FOREIGN KEY(CATEGORY_ID) REFERENCES CATEGORY(CATEGORY_ID)
);

CREATE TABLE CUSTOMERS (
CUSTOMER_ID INTEGER AUTO_INCREMENT,
FIRST_NAME VARCHAR(50) NOT NULL,
LAST_NAME VARCHAR(50) NOT NULL,
EMAIL VARCHAR(255) NOT NULL CHECK(EMAIL LIKE '%@%.%'),
PASSWORD VARCHAR(4000) NOT NULL,
CONSTRAINT CUSTOMER_PK PRIMARY KEY(CUSTOMER_ID)
);

CREATE TABLE ORDERS (
ORDER_ID INTEGER AUTO_INCREMENT,
CUSTOMER_ID INTEGER NOT NULL,
ORDER_DATE DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
TOTAL_AMOUNT DECIMAL(9,2) NOT NULL CHECK(TOTAL_AMOUNT > 0),
CONSTRAINT ORDER_PK PRIMARY KEY(ORDER_ID),
CONSTRAINT ORDER_CUSTOMER_FK FOREIGN KEY(CUSTOMER_ID) REFERENCES CUSTOMERS(CUSTOMER_ID)
);

CREATE TABLE ORDER_DETAILS (
ORDER_DETAILS_ID INTEGER AUTO_INCREMENT,
ORDER_ID INTEGER NOT NULL,
PRODUCT_ID INTEGER NOT NULL,
QUANTITY INTEGER NOT NULL CHECK(QUANTITY >= 0),
UNIT_PRICE DECIMAL(9,2) NOT NULL CHECK(UNIT_PRICE > 0),
CONSTRAINT ORDER_DETAILS_PK PRIMARY KEY(ORDER_DETAILS_ID),
CONSTRAINT ORDER_DETAILS_ORDER_FK FOREIGN KEY(ORDER_ID) REFERENCES ORDERS(ORDER_ID),
CONSTRAINT ORDER_DETAILS_PRODUCT_FK FOREIGN KEY(PRODUCT_ID) REFERENCES PRODUCTS(PRODUCT_ID)
);
```
