## 2. Relationships
- One-to-Many: between Category and Products
- One-to-Many: between Customers and Orders
- Many-to-Many: between Products and Orders via Order_Details

## 3. DDL

```
CREATE TABLE CATEGORY (
CATEGORY_ID INTEGER AUTO_INCREMENT,
CATEGORY_NAME VARCHAR(255) NOT NULL,
CONSTRAINT CATEGORY_PK PRIMARY KEY (CATEGORY_ID)
);

CREATE TABLE PRODUCTS (
PRODUCT_ID INTEGER AUTO_INCREMENT,
CATEGORY_ID INTEGER NOT NULL,
NAME VARCHAR(255) NOT NULL,
DESCRIPTION VARCHAR(4000) NOT NULL,
PRICE DECIMAL(9,2) NOT NULL CHECK(PRICE > 0),
STOCK_QUANTITY INTEGER NOT  NULL CHECK(STOCK_QUANTITY >= 0),
CONSTRAINT PRODUCT_PK PRIMARY KEY(PRODUCT_ID),
CONSTRAINT PRODUCT_CATEGORY_FK FOREIGN KEY(CATEGORY_ID) REFERENCES CATEGORY(CATEGORY_ID)
);

CREATE TABLE CUSTOMERS (
CUSTOMER_ID INTEGER AUTO_INCREMENT,
FIRST_NAME VARCHAR(50) NOT NULL,
LAST_NAME VARCHAR(50) NOT NULL,
EMAIL VARCHAR(255) NOT NULL CHECK(EMAIL LIKE '%@%.%'),
PASSWORD VARCHAR(4000) NOT NULL,
CONSTRAINT CUSTOMER_PK PRIMARY KEY(CUSTOMER_ID)
);

CREATE TABLE ORDERS (
ORDER_ID INTEGER AUTO_INCREMENT,
CUSTOMER_ID INTEGER NOT NULL,
ORDER_DATE DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
TOTAL_AMOUNT DECIMAL(9,2) NOT NULL CHECK(TOTAL_AMOUNT > 0),
CONSTRAINT ORDER_PK PRIMARY KEY(ORDER_ID),
CONSTRAINT ORDER_CUSTOMER_FK FOREIGN KEY(CUSTOMER_ID) REFERENCES CUSTOMERS(CUSTOMER_ID)
);

CREATE TABLE ORDER_DETAILS (
ORDER_DETAILS_ID INTEGER AUTO_INCREMENT,
ORDER_ID INTEGER NOT NULL,
PRODUCT_ID INTEGER NOT NULL,
QUANTITY INTEGER NOT NULL CHECK(QUANTITY >= 0),
UNIT_PRICE DECIMAL(9,2) NOT NULL CHECK(UNIT_PRICE > 0),
CONSTRAINT ORDER_DETAILS_PK PRIMARY KEY(ORDER_DETAILS_ID),
CONSTRAINT ORDER_DETAILS_ORDER_FK FOREIGN KEY(ORDER_ID) REFERENCES ORDERS(ORDER_ID),
CONSTRAINT ORDER_DETAILS_PRODUCT_FK FOREIGN KEY(PRODUCT_ID) REFERENCES PRODUCTS(PRODUCT_ID)
);
```

## 4. SQL Challenges

### 4.1  Write a SQL query to generate a daily report of the total revenue for a specific date.
```
SELECT SUM(TOTAL_AMOUNT) AS REVENUE, DATE(ORDER_DATE) AS DATE
FROM ORDERS
GROUP BY DATE
ORDER BY DATE DESC;
```

### 4.2  Write a SQL query to generate a monthly report of the top-selling products in a given month.

#### 4.2.1 Get each product revenue per month
```
SELECT p.NAME AS PRODUCT_NAME, PRODUC_MOUNTHLY_REVENUE.REVENUE AS MONTHLY_REVENUE, PRODUC_MOUNTHLY_REVENUE.MONTH AS MONTH_PER_YEAR
FROM(
	SELECT od.PRODUCT_ID AS PRODUCT_ID, SUM(od.QUANTITY * od.UNIT_PRICE) AS REVENUE, DATE_FORMAT(ORDER_DATE, '%Y-%m') AS MONTH
	FROM ORDERS o
	JOIN ORDER_DETAILS od
	ON o.ORDER_ID = od.ORDER_ID 
	GROUP BY MONTH, od.PRODUCT_ID 
) AS PRODUC_MOUNTHLY_REVENUE
JOIN PRODUCTS p
    ON PRODUC_MOUNTHLY_REVENUE.PRODUCT_ID = p.PRODUCT_ID
ORDER BY PRODUC_MOUNTHLY_REVENUE.MONTH DESC, PRODUC_MOUNTHLY_REVENUE.REVENUE DESC;
```

#### 4.2.2 Get top-selling product per month
```
SELECT PRODUCT_NAME, MONTHLY_REVENUE, MONTH_PER_YEAR
FROM (
	SELECT 
		p.NAME AS PRODUCT_NAME, 
		PRODUC_MOUNTHLY_REVENUE.REVENUE AS MONTHLY_REVENUE, 
		PRODUC_MOUNTHLY_REVENUE.MONTH AS MONTH_PER_YEAR,
		RANK() OVER (
			PARTITION BY PRODUC_MOUNTHLY_REVENUE.MONTH
			ORDER BY PRODUC_MOUNTHLY_REVENUE.REVENUE DESC
		) AS r
	FROM(
		SELECT od.PRODUCT_ID AS PRODUCT_ID, SUM(od.QUANTITY * od.UNIT_PRICE) AS REVENUE, DATE_FORMAT(ORDER_DATE, '%Y-%m') AS MONTH
		FROM ORDERS o
		JOIN ORDER_DETAILS od
		ON o.ORDER_ID = od.ORDER_ID 
		GROUP BY MONTH, od.PRODUCT_ID 
	) AS PRODUC_MOUNTHLY_REVENUE
	JOIN PRODUCTS p
	    ON PRODUC_MOUNTHLY_REVENUE.PRODUCT_ID = p.PRODUCT_ID
) AS RANKED
WHERE r = 1
ORDER BY MONTH_PER_YEAR DESC;
```

### 4.3 Write a SQL query to retrieve a list of customers who have placed orders totaling more than $500 in the past month. Include customer names and their total order amounts.
```
SELECT CONCAT(FIRST_NAME, ' ', LAST_NAME) AS CUSTOMER_NAME,  SUM(o.TOTAL_AMOUNT) AS TOTAL_ORDERS_AMOUNT, o.ORDER_DATE AS DATE
FROM CUSTOMERS c
JOIN ORDERS o
ON c.CUSTOMER_ID = o.CUSTOMER_ID 
WHERE MONTH(o.ORDER_DATE) = MONTH(CURRENT_DATE - INTERVAL 1 MONTH)
  AND YEAR(o.ORDER_DATE) = YEAR(CURRENT_DATE)
 GROUP BY c.CUSTOMER_ID, o.ORDER_DATE 
HAVING SUM(o.TOTAL_AMOUNT) >= 500
ORDER BY o.ORDER_DATE;
```
